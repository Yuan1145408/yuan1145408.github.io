Node.js 入门

安装

  • 到 nodejs.org 安装 LTS。
  • 检查版本：node -v、npm -v。

Hello, HTTP

// server.js
const http = require('http')
const server = http.createServer((req,res)=>{
  res.setHeader('Content-Type','application/json')
  res.end(JSON.stringify({ ok: true, path: req.url }))
})
server.listen(3000, ()=>console.log('http://localhost:3000'))

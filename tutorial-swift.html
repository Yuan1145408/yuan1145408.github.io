网络请求（URLSession）

import Foundation
let url = URL(string: "https://httpbin.org/json")!
let task = URLSession.shared.dataTask(with: url){ data, _, _ in
  let text = String(data: data!, encoding: .utf8)!
  print(text)
}
task.resume()
RunLoop.main.run()
